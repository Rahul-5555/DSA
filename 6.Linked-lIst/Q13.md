
var rotateRight = function (head, k) {
  //subProblem1: step1: corner cases head is null or has only one node

  if (!head || !head.next) return head; // list empty or has only one node

  //subProblem2: calculate length of linked list
  let length = 0;
  let curr = head;
  while (curr) {
    curr = curr.next;
    length++;
  }
  //subProblem3: to avoid unnecessary rotation Question:
  // Agar list 5 nodes ki hai aur k = 12 ?
  // ðŸ‘‰ 12 % 5 = 2 (same result)
  k = k % length;
  if(k===0) return head; // no rotation needed

  //subProblem4: use two pointer approach to find the new head and tail
  let s = head; // slow pointer
  let f = head; // fast pointer
  // move f pointer to k step ahead
  for (i = 0; i < k; i++) {
    f = f.next;
  }

  //subProblem5: reaching the end of the list
  while (f.next) {
    s = s.next;
    f = f.next;
  }
  // now f is at last node and s is at new tail
  f.next = head;
  let newHead = s.next;
  s.next = null;
  return newHead;
}
